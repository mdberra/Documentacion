import { Injectable } from '@angular/core';
@Injectable()
export class RopaService {
	private nombrePrenda = 'Pantalones';
	private collecionRopa = ['Remera', 'Short', 'Buzo'];

	getPrenda() {
		return this.nombrePrenda;
	}
	addRopa(prenda:string):Array<string> {
		this.collecionRopa.push(prenda);
		return this.getRopa();
	}
	getRopa():Array<string> {
		return this.collecionRopa;
	}
	delRopa(index:number):Array<string> {
		this.collecionRopa.splice(index, 1);
		return this.getRopa();
	}
}
